#include <iostream>
#include <vector>
#include <queue>
using namespace std;

class Graph {
    int V;
    vector<vector<int>> adj;

public:
    Graph(int V) {
        this->V = V;
        adj.resize(V);
    }

    // Undirected graph
    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // -------- DFS (Recursive AI Version) --------
    bool DFS(int node, int goal, vector<bool> &visited, vector<int> &parent) {
        visited[node] = true;

        if (node == goal)
            return true;

        for (int neighbor : adj[node]) {
            if (!visited[neighbor]) {
                parent[neighbor] = node;
                if (DFS(neighbor, goal, visited, parent))
                    return true;
            }
        }
        return false;
    }

    void runDFS(int start, int goal) {
        vector<bool> visited(V, false);
        vector<int> parent(V, -1);

        if (DFS(start, goal, visited, parent)) {
            cout << "DFS Path: ";
            printPath(start, goal, parent);
        } else {
            cout << "Goal not found\n";
        }
    }

    // -------- BFS (AI Version - Shortest Path) --------
    void runBFS(int start, int goal) {
        vector<bool> visited(V, false);
        vector<int> parent(V, -1);
        queue<int> q;

        visited[start] = true;
        q.push(start);

        while (!q.empty()) {
            int node = q.front();
            q.pop();

            if (node == goal) {
                cout << "BFS Shortest Path: ";
                printPath(start, goal, parent);
                return;
            }

            for (int neighbor : adj[node]) {
                if (!visited[neighbor]) {
                    visited[neighbor] = true;
                    parent[neighbor] = node;
                    q.push(neighbor);
                }
            }
        }

        cout << "Goal not found\n";
    }

    // -------- Path Printing --------
    void printPath(int start, int goal, vector<int> &parent) {
        vector<int> path;

        for (int v = goal; v != -1; v = parent[v]) {
            path.push_back(v);
        }

        for (int i = path.size() - 1; i >= 0; i--) {
            cout << path[i] << " ";
        }
        cout << endl;
    }
};

int main() {
    int V, E;

    cout << "Enter number of vertices: ";
    cin >> V;

    cout << "Enter number of edges: ";
    cin >> E;

    Graph g(V);

    cout << "Enter edges (u v):\n";
    for (int i = 0; i < E; i++) {
        int u, v;
        cin >> u >> v;

        if (u >= V || v >= V || u < 0 || v < 0) {
            cout << "Invalid edge!\n";
            i--; // retry
            continue;
        }

        g.addEdge(u, v);
    }

    int start, goal;
    cout << "Enter start node: ";
    cin >> start;

    cout << "Enter goal node: ";
    cin >> goal;

    cout << "\n--- Results ---\n";
    g.runDFS(start, goal);
    g.runBFS(start, goal);

    return 0;
}
